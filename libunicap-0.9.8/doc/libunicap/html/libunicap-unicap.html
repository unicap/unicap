      char category[128];

      char **relations;
      int relations_count;	
      
      union
	    double value;
	    char menu_item[128];

      double stepping;
      
      unicap_property_type_enum_t type;

      unicap_property_flags_t flags_mask;
      
      void *property_data; 
      size_t property_data_size;	
} unicap_property_t;
